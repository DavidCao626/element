## Install
```shell
npm install element-ui-x -S
```

## Quick Start
``` javascript
import Vue from 'vue'
import Element from 'element-ui-x'

Vue.use(Element)

// or
import {
  Select,
  Button
  // ...
} from 'element-ui-x'

Vue.component(Select.name, Select)
Vue.component(Button.name, Button)
```
